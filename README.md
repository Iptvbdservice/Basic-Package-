            var currentPage = 0;
            var itemsPerPage = 3; 

            function renderTestimonials() {
                testimonialGrid.innerHTML = ''; // Clear existing testimonials
                var totalPages = Math.ceil(reviews.length / itemsPerPage);
                if (totalPages === 0) return;
                
                var startIndex = (currentPage * itemsPerPage) % reviews.length;
                
                var pageReviews = [];
                for(var i = 0; i < itemsPerPage; i++){
                    pageReviews.push(reviews[(startIndex + i) % reviews.length]);
                }

                // Using traditional string concatenation for older browser support
                pageReviews.forEach(function(review) {
                    var card = document.createElement('div');
                    card.className = 'testimonial-card p-8 rounded-lg flex flex-col';
                    
                    var starsHTML = '';
                    for (var i = 0; i < 5; i++) {
                        starsHTML += '<svg class="w-5 h-5" fill="currentColor" viewBox="0 0 20 20"><path d="M9.049 2.927c.3-.921 1.603-.921 1.902 0l1.07 3.292a1 1 0 00.95.69h3.462c.969 0 1.371 1.24.588 1.81l-2.8 2.034a1 1 0 00-.364 1.118l1.07 3.292c.3.921-.755 1.688-1.54 1.118l-2.8-2.034a1 1 0 00-1.175 0l-2.8 2.034c-.784.57-1.838-.197-1.539-1.118l1.07-3.292a1 1 0 00-.364-1.118L2.98 8.72c-.783-.57-.38-1.81.588-1.81h3.461a1 1 0 00.951-.69l1.07-3.292z"></path></svg>';
                    }
                    
                    card.innerHTML = '<div class="flex-grow">' +
                        '<div class="flex items-center mb-4">' +
                        '<img src="https://placehold.co/48x48/FACC15/111827?text=' + review.avatar + '" alt="User Avatar" class="w-12 h-12 rounded-full mr-4 border-2 border-yellow-400">' +
                        '<div>' +
                        '<h4 class="font-bold text-white text-lg">' + review.name + '</h4>' +
                        '<div class="flex text-yellow-400">' + starsHTML + '</div>' +
                        '</div>' +
                        '</div>' +
                        '<p class="text-gray-400">"' + review.text + '"</p>' +
                        '</div>';

                    testimonialGrid.appendChild(card);
                });
                currentPage = (currentPage + 1) % totalPages;
            }

            renderTestimonials(); // Initial render
            setInterval(renderTestimonials, 7000); // Rotate every 7 seconds


            // User Count Animator
            var userCountElement = document.getElementById('user-count');
            if (userCountElement) {
                var userCount = 999850; // Starting count
                userCountElement.textContent = userCount.toLocaleString('bn-BD') + '+';

                var updateCount = function() {

                        userCountElement.textContent = userCount.toLocaleString('bn-BD') + '+';
                    } catch (e) {
                        userCountElement.textContent = userCount.toString() + '+';
                    }
                };

                setInterval(updateCount, 2000); // Update every 2 seconds
            }
